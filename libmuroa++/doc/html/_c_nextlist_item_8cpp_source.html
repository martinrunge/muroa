00041 CNextlistItem::CNextlistItem(CRootItem *root_item, std::string text, CCategoryItem*  parent, int posInParent) throw(ExMalformedPatch)
00042 : IContentItem( root_item, parent, CItemType::E_NEXTLISTITEM )
00043 {
00044         m_text = text;
00045         size_t lpos, rpos;
00046         lpos = 0;
00047     m_hash = m_next_free_id++;
00048         m_root_item->setContentPtr(CItemType(CItemType::E_NEXTLISTITEM), this, m_hash );
00049 
00050         try {
00051                 rpos = m_text.find('\t', lpos);
00052                 if( rpos != string::npos ) {
00053                         string mediaitemHashStr = text.substr(lpos, rpos - lpos);
00054                         m_mediaitem_hash = CUtils::str2uint32(mediaitemHashStr.c_str());
00055                         lpos = rpos + 1;
00056                 }
00057                 else {
00058                         throw ExMalformedPatch("error parsing 'media item hash' field, terminating tab char is missing.", -1);
00059                 }
00060 
00061                 string playlistIdStr;
00062                 rpos = m_text.find('\t', lpos);
00063                 if( rpos != string::npos ) {
00064                         playlistIdStr = text.substr(lpos, rpos - lpos);
00065                         lpos = rpos + 1;
00066                 }
00067                 else {
00068                         playlistIdStr = text.substr(lpos);
00069                 }
00070                 m_playlistitem_hash = CUtils::str2uint32(playlistIdStr.c_str());
00071 
00072                 assembleText();
00073 
00074                 if(m_parent) {
00075                         m_parent->addChild(this, posInParent);
00076                 }
00077                 // m_hash is an index that does not change during livetime of this object any more.
00078         }
00079         catch(std::invalid_argument& ex)
00080         {
00081                 throw ExMalformedPatch(ex.what(), -1);
00082         }
00083 }
